## Install

```bash
npm install --save poc-core-system
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'poc-core-system'
import 'poc-core-system/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
